$isApi = (strpos($accept, 'application/json') !== false) || (strpos($_SERVER['REQUEST_URI'], '/api/') === 0);

// Send 503 Service Unavailable + Retry-After (seconds)

header('Retry-After: 120'); // client should retry after 120 seconds (sesuaikan)

        'code' => 503,
